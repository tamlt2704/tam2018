1 Download sample database

curl http://www.postgresqltutorial.com/wp-content/uploads/2017/10/dvdrental.zip --output dvdrental.zip

2 Load into postgresql server

2.1 create database

su - postgres
unzip dvdrental.zip
createdb dvdrental

2.2 import database to server

pg_restore -U postgres -d dvdrental dvdrental.zip --password

3.3 Select Distinct

3.3.1 Create new table

CREATE TABLE t1 (
id serial NOT NULL PRIMARY KEY,
bcolor VARCHAR,
fcolor VARCHAR
);

3.3.2 Insert some data

INSERT INTO t1 (bcolor, fcolor)
VALUES
 ('red', 'red'),
 ('red', 'red'),
 ('red', NULL),
 (NULL, 'red'),
 ('red', 'green'),
 ('red', 'blue'),
 ('green', 'red'),
 ('green', 'blue'),
 ('green', 'green'),
 ('blue', 'red'),
 ('blue', 'green'),
 ('blue', 'blue');

3.3.3 Select data from t1

SELECT 
id,
bcolor,
fcolor
FROM 
t1;

3.3.4 Select distinct bcolor from t1 and order by bcolor

SELECT DISTINCT 
bcolor 
FROM 
t1
ORDER BY
bcolor;

3.3.5 Select distinct on multiple colums

SELECT DISTINCT
    bcolor, fcolor
FROM 
    t1
ORDER BY 
    bcolor, fcolor;

3.3.6 Select distinct bcolor, fcolor from t1 and for each group of duplicates, keep the first row.

SELECT DISTINCT ON 
    (bcolor) bcolor,
    fcolor
FROM 
    t1
ORDER BY
    bcolor, fcolor;

4 Filtering data

4.1 Where

4.1.1 get all customers whose first names are 'Jamie'

SELECT 
    last_name, first_name
FROM
    customer
WHERE 
    first_name = 'Jamie';

4.1.2 Select the customer whose first name is 'Jamie' and last names is 'Rice'

SELECT 
    last_name, first_name
FROM
    customer
WHERE 
    first_name = 'Jamie';
    AND 
    last_name = 'Rice';

4.1.3 find out who paid the rental with amount is either less than 1USD or greater than 8USD

SELECT 
    customer_id, amount, payment_date
FROM 
    payment
WHERE 
    amount <=1 or amount >= 8;

4.2 Limit

4.2.1 Get the first 5 films ordered by filmid, you use the following query:

SELECT
    film_id, title, release_year
FROM 
    film
ORDER BY 
    film_id
LIMIT 5;

4.2.2 Retrieve 4 films starting from the third one ordered by filmid

SELECT 
    film_id, title, release_year
FROM 
    film
ORDER BY
    film_id
LIMIT 4 OFFSET 3;

4.2.3 10 most expensive films

SELECT
    film_id, title, rental_rate
FROM 
    film
ORDER BY
    rental_rate DESC
LIMIT 10;

4.3 Fetch

4.3.1 Get the first row of the film sorted by title
